@app.route('/api/vps/check', methods=['POST'])
def check_vps():
    """Check VPS connection"""
    try:
        vps = VPSConnection.query.first() or VPSConnection()
        
        # Try to ping the VPS
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(5)
        result = sock.connect_ex((vps.vps_ip, vps.vps_port))
        sock.close()
        
        is_connected = result == 0
        vps.is_connected = is_connected
        vps.last_checked = datetime.utcnow()
        
        db.session.add(vps)
        db.session.commit()
        
        return jsonify({'connected': is_connected}), 200
    except Exception as e:
        return jsonify({'error': str(e)}), 500

def get_pending_nodes():
    """Get pending node approvals"""
    config = OpenClawConfig.query.first()
    return jsonify({'pending': config.pending_nodes or []}) if config else jsonify({'pending': []})

@app.route('/api/nodes/approve', methods=['POST'])
def approve_node():
    """Approve a pending node"""
    data = request.json
    node_id = data.get('node_id')
    config = OpenClawConfig.query.first()
    
    if config and config.pending_nodes:
        pending = config.pending_nodes
        if node_id in pending:
            pending.remove(node_id)
            config.nodes_approved += 1
            config.pending_nodes = pending
            db.session.commit()
            return jsonify({'success': True, 'message': 'Node approved'}), 200
    
    return jsonify({'error': 'Node not found'}), 404
